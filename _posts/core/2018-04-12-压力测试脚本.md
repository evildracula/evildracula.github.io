sub myRand ($){
    my $num_len = $_[0];
    my $rnd_num=int(rand(10**$num_len - 1));
    if ($rnd_num<0) {
        $rnd_num++;
    }
    my $num_str="".$rnd_num;
    my $zero_append_num = $num_len - length($num_str);
    if ($zero_append_num > 0) {
        for (my $i=0;$i<$zero_append_num;$i++) {
            $num_str="0".$num_str;
